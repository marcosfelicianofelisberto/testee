function renderizar() {
  const container = document.getElementById("conteudo");
  container.innerHTML = "";

  const categorias = [...new Set(dados.map(d => d.categoria))];

  categorias.forEach(cat => {
    const titulo = `<h2>${cat}</h2>`;
    let tabela = `<table><tr><th>Produto</th><th>Quantidade</th></tr>`;

    dados
      .filter(d => d.categoria === cat)
      .forEach((item, i) => {
        const index = dados.indexOf(item);
        tabela += `
        <tr>
          <td contenteditable oninput="editar(${index}, 'produto', this.innerText)">
            ${item.produto}
          </td>
          <td contenteditable oninput="editar(${index}, 'quantidade', this.innerText)">
            ${item.quantidade}
          </td>
        </tr>`;
      });

    tabela += "</table>";
    container.innerHTML += titulo + tabela;
  });
}

function editar(index, campo, valor) {
  dados[index][campo] = valor;
}

function salvar() {
  localStorage.setItem("estoque", JSON.stringify(dados));
  alert("Estoque salvo!");
}
